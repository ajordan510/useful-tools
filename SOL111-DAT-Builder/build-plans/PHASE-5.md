### Assembly Logic Details

#### Executive Control Section

```javascript
generateExecutiveControl(state) {
  const lines = [];
  lines.push(NastranCards.comment('=' .repeat(64)));
  lines.push(NastranCards.comment(`SOL 111 Frequency Response Analysis`));
  lines.push(NastranCards.comment(`Generated by SOL 111 .dat Builder`));
  lines.push(NastranCards.comment(`Date: ${new Date().toISOString().slice(0,10)}`));
  if (state.analyst) lines.push(NastranCards.comment(`Analyst: ${state.analyst}`));
  if (state.jobName) lines.push(NastranCards.comment(`Job: ${state.jobName}`));
  lines.push(NastranCards.comment('=' .repeat(64)));

  // Optional NASTRAN statement
  if (state.nastranVersion === 'msc') {
    lines.push('NASTRAN SYSTEM(316)=19');  // Suppress DMAP listing
  }

  lines.push(`ID SOL111,BUILDER`);
  lines.push('SOL 111');
  lines.push('TIME 600');
  lines.push('CEND');

  return lines;
}
```

#### Case Control Section

```javascript
generateCaseControl(state) {
  const lines = [];
  lines.push(NastranCards.comment('=' .repeat(64)));
  lines.push(NastranCards.comment('CASE CONTROL SECTION'));
  lines.push(NastranCards.comment('=' .repeat(64)));

  if (state.title) lines.push(`TITLE = ${state.title}`);
  if (state.subtitle) lines.push(`SUBTITLE = ${state.subtitle}`);
  lines.push('ECHO = NONE');
  lines.push(`SPC = ${state.spcSetId}`);
  lines.push(`METHOD = ${state.eigrl.sid}`);

  // SET definitions
  for (const set of state.sets) {
    lines.push(...NastranCards.formatSET(set));
  }

  // Subcases
  for (const sc of state.subcases) {
    lines.push(`SUBCASE ${sc.id}`);
    lines.push(`  SUBTITLE = ${sc.label}`);
    lines.push(`  DLOAD = ${sc.dloadSid}`);

    // Output requests for this subcase
    const outputs = state.sameOutputsAllSubcases
      ? state.outputRequests
      : (sc.outputRequests || state.outputRequests);

    for (const [type, cfg] of Object.entries(outputs)) {
      if (!cfg.enabled) continue;
      const nastranName = outputTypeToCardName(type);
      const options = `${cfg.sort},${cfg.format}`;
      const setRef = cfg.set === 'ALL' ? 'ALL' : cfg.set;
      lines.push(`  ${nastranName}(${options}) = ${setRef}`);
    }
  }

  return lines;
}
```

#### Bulk Data Section

```javascript
generateBulkData(state) {
  const lines = [];
  lines.push(NastranCards.comment('=' .repeat(64)));
  lines.push(NastranCards.comment('BULK DATA SECTION'));
  lines.push(NastranCards.comment('=' .repeat(64)));
  lines.push('BEGIN BULK');

  // INCLUDE statement for model BDF
  if (state.bdfFilePath) {
    lines.push(`INCLUDE '${state.bdfFilePath}'`);
  }

  // Modal extraction
  lines.push(NastranCards.comment('---- Modal Extraction ----'));
  lines.push(...NastranCards.formatEIGRL(state.eigrl));

  // Frequency definition
  lines.push(NastranCards.comment('---- Frequency Definition ----'));
  lines.push(...formatFrequencyCards(state));

  // Damping
  lines.push(NastranCards.comment('---- Damping ----'));
  lines.push(...formatDampingCards(state));

  // Dynamic loads (per subcase)
  lines.push(NastranCards.comment('---- Dynamic Loads ----'));
  for (const sc of state.subcases) {
    lines.push(...formatSubcaseLoads(sc, state));
  }

  // Parameters
  lines.push(NastranCards.comment('---- Parameters ----'));
  if (state.generatePch) lines.push(...NastranCards.formatPARAM('POST', -1));
  if (state.generateOp2) lines.push(...NastranCards.formatPARAM('POST', 0));
  lines.push(...NastranCards.formatPARAM('GRDPNT', 0));

  // Pre-analysis check cards
  lines.push(...formatPreCheckCards(state));

  lines.push('ENDDATA');
  return lines;
}
```

### SID Management

All Nastran SID (Set ID) references must be unique and consistent between Case Control and Bulk Data. The assembler assigns SIDs automatically:

```javascript
const SIDManager = {
  _next: 100,  // Start from 100 to avoid conflicts with model BDF
  reset() { this._next = 100; },
  next() { return this._next++; },

  // Pre-assigned ranges:
  // EIGRL: 1-9
  // FREQ:  10-19
  // TABDMP1: 20-29
  // DLOAD/RLOAD: 100+ (auto-incremented)
  // TABLED1: 200+ (auto-incremented)
  // DAREA: 300+ (auto-incremented)
  // SET: 900+ (auto-incremented)
};
```

The assembler runs SID assignment as a first pass before generating any cards, ensuring all cross-references are consistent.

### Preview Features

- **Read-only mode** (default): `<pre>` block with syntax highlighting
- **Edit mode** (toggle): `<textarea>` for manual modifications
- **Syntax highlighting** (read-only mode only):
  - Comment lines (`$`): green (`--success` or dedicated `--comment` color)
  - Card names (first 8 chars of non-comment lines): bold blue
  - Continuation markers (`+`): gray
  - `BEGIN BULK`, `ENDDATA`, `CEND`: bold
  - Numbers: default color
- **Line numbers**: displayed in a gutter alongside the preview
- **Statistics**: line count, card count, estimated file size

### Edit Mode

- Toggle button switches between `<pre>` (highlighted) and `<textarea>` (editable)
- When switching to edit: copy highlighted text to textarea
- When switching back to preview: re-highlight from textarea content
- **Important**: Manual edits are one-way. The textarea content is what gets downloaded, not the State-derived version. A "Regenerate" button discards edits and re-generates from current State.

### Download

```javascript
function downloadDat() {
  const content = State.editedDatText || State.generatedDatText;
  const filename = `${State.jobName || 'sol111'}.dat`;
  const blob = new Blob([content], { type: 'text/plain' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}
```

### Copy to Clipboard

```javascript
function copyToClipboard() {
  const content = State.editedDatText || State.generatedDatText;
  navigator.clipboard.writeText(content).then(() => {
    setStatus('Copied to clipboard', 'success');
  });
}
```

3. Line numbers displayed alongside the preview
4. Statistics (line count, card count) shown below preview
